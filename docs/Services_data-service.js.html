/**
 * A function to generate a fetch call to the dummyCardsData.json file which is present in the public directory, 
 * to retrieve the json data. The json data has content which is passed to the Landingpage.js file and rendered 
 * in each card.
 */

export const getProjects = () => {
    return fetch("./dummyCardsData.json")
    .then((res)=>res.json())
}
