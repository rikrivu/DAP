import React from 'react';
import Nimbus from './Pages/NIMBUS/nimbus';
import Landingpage from './Pages/LandingPage/landingpage';
import { BrowserRouter, Route, Switch } from "react-router-dom";
import About from './Pages/About/about';
import ContactUs from './Pages/ContactUs/contactus';
import Infrastructure from './Pages/Infrastructure/infrastructure';
import otherTools from './Pages/Other tools/otherTools'

/**
 * React Router component for this project
 * Desribes all the internal routes for the React application. All of these routes can be used
 * to navigate between the components defined in this React application.
 * @returns The route to that particular webpage
 */


function Router1() {
    return (
        <div>
            <BrowserRouter>
                <Switch>
                    <Route path="/" exact component={Landingpage}></Route>
                    <Route path="/datamining" component={Nimbus}></Route>
                    <Route path="/about" component={About}></Route>
                    <Route path="/contactus" component={ContactUs}></Route>
                    <Route path="/infrablogs"  component={Infrastructure}></Route>
                    <Route path="/othertools"  component={otherTools}></Route>
                </Switch>
             </BrowserRouter>
        </div>
    )
    
}

export default Router1
