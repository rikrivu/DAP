import React, { useState, useEffect } from 'react';
import Header from '../../Components/Shared/Header/Header';
import Footer from '../../Components/Shared/Footer/Footer';
import FileInput from '../../Components/Nimbus/FileInput';
import Output from '../../Components/Nimbus/Output';
import 'bootstrap/dist/css/bootstrap.min.css';
import { Container, Row,  Col } from 'react-bootstrap';
import './nimbus.css';
import cardOutline from '../../Components/Nimbus/cardOutline';

/**
 * Describes the outline for the nimbus EDA page. This component describes what other
 * components must be included and where each of these components are defined or located
 * on the webpage.
 * @returns A visual representation of nimbus EDA webpage
 */

function Nimbus() {
// Creating hook state to store the filename received from the backend API
const [file, setFile] = useState(null);
// Creating hook state to store the dataframes received from the backend API
const [dataFrames, setdataFrames] = useState(null);
// Creating hook state to store the lists of data received from the backend API
const [dataLists, setdataLists]=useState({"":""});
// Creating hook state to store the loading state
const [loading, setLoading] = useState(false);
const [imageExists, setImageExists] = useState(false);

// Created a callback function to receive data from the child and store the data in respective states
const callbackFunction = (childData) => {
    console.log(childData)
    setdataFrames(childData.dataFrame);
    setFile(childData["filename"]);
    setdataLists(childData.list);
    setImageExists(childData.Image_exists)
}

// Created a callback function to receive data from the child and store loading data in its respective state
const childcallback = (childData) => {
  setLoading(childData);
}
  
  return (

    <div style={{height:"100%"}}>
      <Header />
      {
        //Using a container to hold all the components
      }
      <Container fluid>
        <Row>
          {
            // Calling the FileInput component
          }
          <Col sm="5" md="5" lg="5" className="column1">
              <div className="card">
              {/* <cardOutline className="card"> */}
                  <FileInput callback={childcallback} parentCallback={callbackFunction}/>
              {/* </cardOutline> */}
              </div>
          </Col>
          <Col sm="7" md="7" lg="7">
              
          {
            // Output Window to display the results
          }
              <div className="card1">
              {/* <cardOutline className="card1"> */}
                  <h3 style={{textAlign: "center", paddingBottom:"5px"}}>Results</h3> 
                  {
                    file==null && loading==false &&
                    <h6 style={{textAlign: "center"}}>No Results to display</h6>
                  }
                  {
                    // Checking for loading and if it is true, rendering loading component
                  }
                  {loading==true &&
                  <div className="alertbox">
                    <img style={{textAlign:"center", marginLeft:"auto", marginRight:"auto"}} alt="Loading" src="https://media.giphy.com/media/3oEjI6SIIHBdRxXI40/giphy.gif"></img>
                  </div>
                  }
                  {
                    // Rendering the Output Component if it gets the correct data from backend API
                  }
                  { file!=null &&
                  <Output image={imageExists} filename={file} tabledata={dataFrames} list={dataLists}/>
                  }
              {/* </cardOutline> */}
              </div>
              
          </Col>
        </Row>
        
        
      </Container>
      {
        //Rendering the Footer component
      }
      <Footer style={{marginTop:"20%"}} className="footer" /> 
      

    </div>
    
    

  );
  

}

export default Nimbus=React.memo(Nimbus);
